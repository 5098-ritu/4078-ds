Code for Question 2



#include < stdio.h >


int fibonacci(int n) {
    
    if (n == 0) {
        return 0;
    } else if (n == 1) {
        return 1;
    }
    
    return fibonacci(n - 1) + fibonacci(n - 2);
}

int main() {
    int num;

    printf("Enter a non-negative integer: ");
    scanf("%d", &num);

    if (num < 0) {
        printf("Fibonacci is not defined for negative numbers.\n");
    } else {
        printf("Fibonacci of %d = %d\n", num, fibonacci(num));
    }

    return 0;
}

Output




Enter a non-negative integer: 12
Fibonacci of 12 = 144

Code for Question 3




#include < stdio.h >

int sumNaturalNumbers(int n) {
    
    if (n == 0) {
        return 0;
    }
    
    return n + sumNaturalNumbers(n - 1);
}

int main() {
    int num = 10;  
    int sum = sumNaturalNumbers(num);
    printf("The sum of natural numbers up to %d is %d\n", num, sum);

    return 0;
}

Output



The sum of natural numbers up to 10 is 55

#include < stdio.h >


void decimalToBinary(int n) {
    // Base case: if n is 0, return
    if (n == 0) {
        return;
    }
    
    decimalToBinary(n / 2);
   
    printf("%d", n % 2);
}

int main() {
    int num;

    printf("Enter a non-negative integer: ");
    scanf("%d", &num);

    if (num < 0) {
        printf("Please enter a non-negative integer.\n");
    } else if (num == 0) {
        printf("Binary representation of 0 is 0\n");
    } else {
        printf("Binary representation of %d is: ", num);
        decimalToBinary(num);
        printf("\n");
    }

    return 0;
}

Output



Enter a non-negative integer: 10
Binary representation of 10 is: 1010

#include < stdio.h >

int findLargest(int arr[], int size) {
   
    if (size == 1) {
        return arr[0];
    }

    
    int largestInRest = findLargest(arr, size - 1);
    return (arr[size - 1] > largestInRest) ? arr[size - 1] : largestInRest;
}

int main() {
    int n;

    printf("Enter the number of elements in the array: ");
    scanf("%d", &n);
    
    int arr[n];
    printf("Enter the elements of the array:\n");
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    int largest = findLargest(arr, n);
    printf("The largest element in the array is: %d\n", largest);

    return 0;
}

int findSmallest(int arr[], int n) {
    
    if (n == 1) {
        return arr[0];

}

Code for Question 7



#include < stdio.h >


int sumOfArray(int arr[], int n) {
    
    if (n == 0) {
        return 0;
    }
   
    return arr[n - 1] + sumOfArray(arr, n - 1);
}


double arithmeticMean(int arr[], int n) {
    if (n == 0) {
        return 0; 
    }
    int sum = sumOfArray(arr, n);
    return (double)sum / n; 
}

int main() {
    int arr[] = {23,12,7,8,4,3};
    int n = sizeof(arr) / sizeof(arr[0]);
    
    double mean = arithmeticMean(arr, n);
    printf("The arithmetic mean of the array is: %.2f\n", mean);
    
    return 0;
}

Output



The arithmetic mean of the array is: 9.50
