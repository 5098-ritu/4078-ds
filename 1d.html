    if (start == end) {
        for (int i = 0; i <= end; i++) {
            printf("%d ", arr[i]);
        }

#include< stdio.h >
int isPalindrome(int num) {
    int reversed = 0;
    int original = num;
    while (num != 0) {
        int remainder = num % 10;
        reversed = reversed * 10 + remainder;
        num /= 10;
    }
    return (original == reversed);
}

void findMinima(int arr[], int size) {
    if (size > 1 && arr[0] < arr[1]) {
        printf("Local minima found at index 0: %d\n", arr[0]);
    }
    for (int i = 1; i < size - 1; i++) {
        if (arr[i] < arr[i - 1] && arr[i] < arr[i + 1]) {

    if (size > 1 && arr[size - 1] < arr[size - 2]) {

}

#include< stdio.h >
int trapWater(int h[], int n) {
    if (n == 0) return 0;

    int l = 0, r = n - 1;
    int left_max = 0, right_max = 0;
    int water = 0;

    while (l < r) {
        if (h[l] < h[r]) {
            if (h[l] >= left_max) {
                left_max = h[l];
            } else {
                water += left_max - h[l];
            }
            l++;
        } else {
            if (h[r] >= right_max) {
                right_max = h[r];
            } else {
                water += right_max - h[r];
            }
                r--;
        }
    }
    return water;   }
    
   int main() {
    int n;
    printf("Enter number of elevation points: ");
    scanf("%d", &n);
    int h[n];
    printf("Enter elevation heights:\n");
    for (int i = 0; i < n; i++) {
        scanf("%d", &h[i]);
    }
    int result = trapWater(h, n);
    printf("Total water trapped: %d\n", result);
    return 0;
}
