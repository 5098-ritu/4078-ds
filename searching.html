                        int linearSearch(char str[], char ch, int index) {

                                return -1;
                            }

                            }

Code for Question 5


                    #include < stdio.h>

                        int binarySearch(int arr[], int size, int value) {
                            int left = 0, right = size - 1;
                        
                            while (left <= right) {
                                int mid = left + (right - left) / 2;
                        
                                if (arr[mid] == value) {
                                    return mid;
                                }
                                if (arr[mid] < value) {
                                    left = mid + 1;
                                } else {
                                    right = mid - 1;
                                }
                            }
                        
                            return -1;
                        }
                        
                        int main() {
                            int n, value;
                        
                            printf("Enter the number of elements in the sorted array: ");
                            scanf("%d", &n);
                        
                            int arr[n];
                            printf("Enter %d sorted integers:\n", n);
                            for (int i = 0; i < n; i++) {
                                scanf("%d", &arr[i]);
                            }
                        
                            printf("Enter the integer value to search: ");
                            scanf("%d", &value);
                        
                            int index = binarySearch(arr, n, value);
                        
                            if (index != -1) {
                                printf("Value %d found at index %d.\n", value, index);
                            } else {
                                printf("Value %d not found in the array.\n", value);
                            }
                        
                            return 0;
                        }

Code for Question 6


                    #include < stdio.h>
                        #include < string.h>
                        
                        int binarySearch(char str[], char ch, int left, int right) {
                            if (left > right) {
                                return -1;
                            }
                            
                            int mid = left + (right - left) / 2;
                        
                            if (str[mid] == ch) {
                                return mid;
                            }
                            if (str[mid] < ch) {
                                return binarySearch(str, ch, mid + 1, right);
                            }
                            return binarySearch(str, ch, left, mid - 1);
                        }
                        
                        int main() {
                            char str[100], ch;
                        
                            printf("Enter a sorted string: ");
                            scanf("%s", str);
                        
                            printf("Enter a character to search: ");
                            scanf(" %c", &ch);
                        
                            int index = binarySearch(str, ch, 0, strlen(str) - 1);
                        
                            if (index != -1) {
                                printf("Character '%c' found at index %d.\n", ch, index);
                            } else {
                                printf("Character '%c' not found in the string.\n", ch);
                            }
                        
                            return 0;
                        }

Code for Question 9


                    #include < stdio.h>

                        int linearSearch(int arr[], int size, int value) {
                            for (int i = 0; i < size; i++) {
                                if (arr[i] == value) {
                                    return i;
                                }
                            }
                            return -1;
                        }
                        
                        int main() {
                            int n, bestValue, worstValue;
                        
                            printf("Enter the number of elements in the array: ");
                            scanf("%d", &n);
                        
                            int arr[n];
                            printf("Enter %d integers:\n", n);
                            for (int i = 0; i < n; i++) {
                                scanf("%d", &arr[i]);
                            }
                        
                            bestValue = arr[0];
                            worstValue = -1;
                        
                            int bestIndex = linearSearch(arr, n, bestValue);
                            if (bestIndex != -1) {
                                printf("Best-case: Value %d found at index %d.\n", bestValue, bestIndex);
                            } else {
                                printf("Best-case: Value %d not found.\n", bestValue);
                            }
                        
                            int worstIndex = linearSearch(arr, n, worstValue);
                            if (worstIndex != -1) {
                                printf("Worst-case: Value %d found at index %d.\n", worstValue, worstIndex);
                            } else {
                                printf("Worst-case: Value %d not found.\n", worstValue);
                            }
                        
                            return 0;
                        }
