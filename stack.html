void sortStack(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                // Swap elements
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

int evaluatePrefix(char prefix[]) {
    int stack[100]; 
    int top = -1;   
    int i; 
    for (i = 0; prefix[i] != '\0'; i++); 
    for (i -= 1; i >= 0; i--) {
       if (prefix[i] == ' ') {
            continue;
        }
     if (prefix[i] >= '0' && prefix[i] <= '9') {
            int num = 0;
            while (i >= 0 && prefix[i] >= '0' && prefix[i] <= '9') {
                num = num * 10 + (prefix[i] - '0');
                i--;
            }
            stack[++top] = num; 
            i++; 
        } else { 
            int operand1 = stack[top--]; 
            int operand2 = stack[top--]; 
            int result;

            switch (prefix[i]) {
                case '+':
                    result = operand1 + operand2;
                    break;
                case '-':
                    result = operand1 - operand2;
                    break;
                case '*':
                    result = operand1 * operand2;
                    break;
                case '/':
                    result = operand1 / operand2;
                    break;
                case '^':
                    result = 1;
                    for (int j = 0; j < operand2; j++) {
                        result *= operand1; 
                    }
                    break;
                default:
                    return 0; 
            }
            stack[++top] = result; 
        }
    }
    return stack[top]; 
}

            prefix[i] = '\0';

    }
}
int pop(struct Stack* s) {
    if (!isEmpty(s)) {
        return s->disks[(s->top)--];
    }
    return -1;
}

void towerOfHanoi(int n, struct Stack* source, struct Stack* auxiliary, struct Stack* target) {
    
    for (int i = n; i >= 1; i--) {
        push(source, i);
    }
    int totalMoves = (1 << n) - 1; // 2^n - 1
    for (int i = 1; i <= totalMoves; i++) {
        struct Stack* from;
        struct Stack* to;
       if (i % 3 == 1) {
            from = source; to = target;
        } else if (i % 3 == 2) {
            from = source; to = auxiliary;
        } else {
            from = auxiliary; to = target;
        }
        if (!isEmpty(from)) {
            int disk = pop(from);
            moveDisk(from == source ? 'A' : (from == auxiliary ? 'B' : 'C'),
                      to == source ? 'A' : (to == auxiliary ? 'B' : 'C'), disk);
            push(to, disk);        }
    }
}
